      # 类型定义

      值、参数、返回值被指定数据类型后

      若被赋值其他类型的数据时，VSCode 会有错误提示

      若直接执行，无论 node 环境还是 eslint 都会有报错：

      # Type 'string' is not assignable to type 'number'.

      # 变量

      const 变量: 类型 = 值;

      如下：

      type person = {
  name: string;
  age: number;
};

const arr: Array<person> = [
  {
    name: "Andy",
    age: 28,
  },
];

      # 函数

      const 函数(参数: 类型): 类型 => {
 return 返回值
}

      如下：

      const fun = (params: number | string): Array<number | string> => {
  return [params];
};

console.log(fun(1)); // [1]
console.log(fun("1")); // ['1']

      # 联合类型

      # |

      表示或，在多个类型中任选其一

      只要属于任意一个即可

      数据类型 | 数据类型;

      let a: number | string;

a = 100;
a = "100";

      # &

      表示且，指定的多个类型必须都要符合

      数据类型 & 数据类型;

      let a: { name: string } & { age: number };

a = { name: "Jack" }; // 报错
a = { name: "Jack", age: 28 };

      # 类型断言 as

      (变量 as 数据类型) < 类型 > 变量;

      let a: unknown;
let b: string = a; // 报错
let c: string = a as string; // 不报错

      # 基础类型

      # number 数值

      let 变量: number = 值;

      let num: number = 10;
num = 20;
num = "20"; // 报错

      TS中不区分整数型和浮点型

      let a: number = 10;
let b: number = 10.0;

      # string 字符串

      let 变量: string = 值;

      let str: string = "hello";
str = "10";
str = 10; // 报错

      # boolean 布尔

      let 变量: boolean = true;
let 变量: boolean = false;

      let flag: boolean;

flag = true;
flag = false;

      # Array 数组

      有两种定义方式：

      1. [变量: 子元素类型[ ]](#类型[ ])
      2. 变量: Array<子元素类型>

      # 类型[ ]

      通过**元素类型[]**指定数组类型

      const 变量: 元素类型[] = [元素, 元素];

      如下：

      // 基础类型元素的数组
const perons: string[] = ["Andy", "Tom", "Jack"];

// 对象元素的数组
type person = {
  name: string;
  age: number;
};
const arr: person[] = [
  {
    name: "Andy",
    age: 28,
  },
  {
    name: "Tom",
    age: 17,
  },
];

      # Array<类型>

      也可通过范型指定数组类型

      const 变量: Array<子元素类型> = [子元素, 子元素];

      如下：

      // 基础类型子元素的数组
const perons: Array<string> = ["Andy", "Tom", "Jack"];

// 对象子元素的数组
type person = {
  name: string;
  age: number;
};
const arr: Array<person> = [
  {
    name: "Andy",
    age: 28,
  },
  {
    name: "Tom",
    age: 17,
  },
];

      # tuple 元组

      元组可以算是数组类型的一种，是指一个固定长度的数组

      区别于数组类型Array<元素类型>，元组的元素个数固定

      赋值的元素个数与数据类型必须与定义类型时一致，否则报错

      const 变量: [元素类型] = [元素];
const 变量: [元素类型, 元素类型] = [元素, 元素];

      如下：

      const tup1: [number] = [1];
const tup1: [number] = [1, 2]; // 报错

const tup2: [string, number] = ["1", 2];

      # enum 枚举

      通过定义表示符，来表示固定的状态或固定的值

      表示符的值为前一个表示符的值 +1

      表示符的值默认从0开始

      enum 枚举名 {
    表示符[= 整数],
    表示符[= 整数],
}

      enum gender {
  male,
  female,
}
console.log(gender.male); // 0
console.log(gender.female); // 1

      可自定义手动赋值枚举中表示符的值

      enum renderStatus {
  error = -1,
  succeed,
  loading,
}
console.log(renderStatus.error); // -1
console.log(renderStatus.succeed); // 0
console.log(renderStatus.loading); // 1

      // 设定 male为1，female为2
enum Gender {
  male = 1,
  female = 2,
}

let jack: { name: string; gender: Gender } = {
  name: "Jack",
  gender: Gender.male,
};
let alice: { name: string; gender: Gender } = {
  name: "Alice",
  gender: Gender.female,
};

      # any

      指定了any类型的话可赋值任何数据类型

      let a: any;
a = 10;
a = "10";
a = true;
a = undefined;
a = null;

      声明变量后不指定数据类型，TS默认指定any

      // 显式 any
let 变量: any;

// 隐式 any
let 变量;

      开发中不要使用 any

      因为 any 类型相当于关闭了TS对该变量的数据类型监测

      不仅变量自身可以被赋值任意数据，并且该变量直接赋值给其他变量时，也会破坏其他规定了类型的变量

      let a: any;
a = 100;
a = "100";
a = true;

let b: string = a; // 居然不报错

      # unknown

      不能确定数据类型时可指定unknown类型

      unknown类型的变量自身可以被随意赋值，效果和andy一样但不能赋值给其他变量，相当于一个类型安全的any

      let a: unknown;
a = "100";
a = 100;

let b: string;
b = a; // 报错
// Type 'unknown' is not assignable to type 'string'.

if (typeof a === "string") b = a;
// 通过判断类型是可以的

      # undefined 和 null

      # void

      表示函数无返回值

      函数若无返回值可以指定void

      const 函数 = (): void => {
  函数体;
};

      如下：

      function func(): void {
  console.log("hello");
}

      const printDate = (): void => {
    console.log(new Date);

      # never 类型

      function showError(): never {
  throw new Error("error");
}

      # 类型别名 type

      除了 TS 提供的基础类型，也可以自定义类型

      type 类型别名 = 类型内容;

      如下：

      type myType = 10 | 20 | 30 | 40;

let num01: myType = 2; // 报错
let num02: myType = 20;

      type Person = { name: string } & { age: number };

let jack: Person = { name: "Jack", age: 28 };

      # 字面量对象

      定义了字面量类型后，不能赋值该字面量以外的值

      变量: 字面量;

      let obj: { name: string; age: number };
obj = {}; //. 报错
obj = { name: "Jack", age: 28 };

      变量: 字面量一 | 字面量二;

      let gender: "male" | "female";

gender = "male";
gender = "female";
gender = "你好"; // 报错

      # 一般对象

      变量: {
  属性: 类型;
}

      let obj: { name: string; gender: "male" | "female" };

      # 可选属性 ？

      在属性名后加上?表示该属性可选

      变量: { 属性?: 类型 }

      let obj: { name: string; age?: number };
obj = { name: "Jack", age: 28 };
obj = { name: "Jack" };

      # 任意属性

      变量: { [propsName: string]: any}

      let obj: { [propsName: string]: any };

obj = { name: "Jack" };
obj = { name: "Jack", age: 28 };
obj = { class: 11, students: 30 };
