# 简介

作用近似于 类型别名 type，但是扩展性比 type 强大，更推荐使用

# 定义

只能声明 类、对象、函数

直接在 { } 里声明，接口名名首字母大写

interface 对象类型接口名 {
  属性: 类型;
  属性?: 类型;
  readonly 属性: 类型;
  函数属性(): 返回值类型;
  函数属性: () => 返回值类型;
}

interface 函数类型接口名 {
  (参数: 类型, 参数: 类型): 返回值类型;
}

// 对象类型接口
interface Person {
  name: string;
  age: number;
  gender: "male" | "female";
}

// 有参数和返回值的函数类型接口
interface A {
  (a: number, b: number): number;
}
// 无参数无返回值的函数类型接口
interface B {
  (): void;
}
const func1: A = (a, b) => a + b;
const func2: B = () => console.log("hello");

# 声明合并

同名接口内容合并

interface Person {
  name: string;
  age: number;
}
interface Person {
  gender: string;
}

const jack: Person = {
  name: "Jack",
  age: 18,
  gender: "male",
};

# ❌extends

接口的继承通过 extends

interface A {
  // 内容
}
interface B extends A {
  // 内容
}

interface A {
  // 内容
}
interface B {
  // 内容
}
interface C extends A, B {
  // 内容
}

type X = 类型内容;
interface A extends X {
  // 内容
}
