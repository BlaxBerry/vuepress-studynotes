                  # 基础类型

                  类型 含义 例子
                  number 数值 整数、小数、正负数、二・八・十六进制、NaN、Infinity、-Infinity
                  string 字符串 "你好"
                  boolean 布尔值 true、false
                  any 任意值
                  unknown 任意值
                  null 空值 null
                  undefined 值未定义 undefined
                  void 无返回值的函数 详见下文
                  never 无返回值的函数 详见下文
                  object 对象 包含键值对的对象、{ }

                  object 类型（不常用）

                  能赋值的数据仅为包含键值对的对象或空对象 {}
                  但无法精确控制键值对的类型，对象类型的定义多通过字面量形式。详见下文

                  let a: object;

a = {};
a = { k: 1, y: 2 };

                  any 类型（不要用）

                  不但影响自己还会祸害别人，开发时坚决不要使用
                  any 类型十分不安全，相当于关闭了 TypeScript 对该变量的类型监测，不仅自身可以被赋值任意类型数据，且将自身直接赋值给其他变量时会破坏那个变量的类型限制，影响其逻辑运算等处理的结果。如下：

                  let a: any;
a = 100;
a = "100";
a = true;

const b: string = a; // 居然不报错！（现在的 a 是 boolean 型而非 string 类型）

                  unknown 类型

                  • 相当于安全的 any 类型
                    仅自己可以被赋值任意类型数据，不能被赋值给 unknown 或 any 以外的类型
                  let a: unknown;
a = 100;
a = "100";
a = true;

const b: unknown = a;
const c: any = a;
const d: string = a; // 报错
const e: number = a; // 报错
                  • 不能作为函数的返回值类型
                  const a = (): any => {};
const b = (): unknown => {}; // 报错，声明类型不为 void 或 any 的函数有必须返回值
                  • 不能使用运算符
                  const func = (x: unknown) => {
  x == 5;
  x !== 10;
  x >= 0; // 报错
  x + 1; // 报错
  x * 2; // 报错
  -x; // 报错
  +x; // 报错
};

                  # 字面量类型

                  定义基础类型时只要数据属于该类型范围内就被赋值

                  若想限定到是哪个数据则要使用字面量定义类型

                  let a: number;
a = 1;
a = 2;
a = 99999;

let b: 1 | 2;
b = 1;
b = 2;
b = 999; // 报错，不能将类型 999 分配给类型 1 | 2

                  # 复杂类型

                  # 数组

                  写法 含义 例子
                  [元素类型, 元素类型] 元素个数固定，类型不同 [string, number]
                  Array<元素类型> 元素个数不固定，类型相同 Array<string>
                  元素类型[ ] 同上 string[ ]
                  /* 个数固定 & 类型不同的数组类型 */
const a: [string, number, boolean] = ["1", 2, true];

/* 个数不固定 & 类型相同的数组类型 */
const b: Array<number> = [1, 2, 3];
const c: number[] = [1, 2, 3];

                  如下：数组元素为复杂类型的场合

                  type Person = {
  name: string;
  age: number;
};

const people: Array<Person> = [
  { name: "Andy", age: 28 },
  { name: "Tom", age: 18 },
];

                  # 对象

                  对象类型的写法：

                  1. object 类型（不推荐） 详见上文
                  2. 字面量
                    1. 直接定义在数据后（不常用）详见下文
                    2. type 类型别名 详见下文
                    3. interface 接口 详见跳转
                  • object 类型 & 字面量写法

                  比起 object 类型，字面量形式写法能精确控制键值对的类型

                  并且比起直接定义到数据后，type 或 interface 能重复使用且代码清晰

                  const 数据: object = 值;

const 数据: { 属性: 类型 } = 值;

type Type别名 = {
  属性: 类型;
};

interface 接口 {
  属性: 类型;
}
                  • 可选属性（ ? ）
                  type Type别名 = {
  属性?: 类型
}
interface 接口 = {
  属性?: 类型
}

                  type Type = {
  a: string;
  b?: string;
};

const obj_1: Type = { a: "A", b: "B" };
const obj_2: Type = { a: "A" }; // 不报错
                  • 只读属性（ readonly ）

                  定义为只读属性的属性无法被修改

                  type Type别名 = {
  readonly 属性: 类型
}
interface 接口 = {
  readonly 属性: 类型
}

                  type Type = {
  a: string;
  readonly b: string;
};

const obj: Type = { a: "A", b: "B" };
obj.a = "AAA";
obj.b = "BBB"; // 报错，无法分配到 b ，因为它是只读属性
                  • 函数属性

                  对象的属性为函数时

                  type Type别名 = {
  函数名(): 返回值类型;
  函数名: () => 返回值类型;
};

interface 接口名 {
  函数名(): 返回值类型;
  函数名: () => 返回值类型;
}

                  interface MyInterface {
  A(): void;
  B: () => void;
}
const xx: MyInterface = {
  A: () => console.log("A"),
  B: () => console.log("B"),
};

xx.A(); // A
xx.B(); // B
                  • 键值对
                  // 固定的键值对
type Type别名 = {
  属性A: 类型;
  属性B: 类型;
};
interface 接口名 {
  属性A: 类型;
  属性B: 类型;
}

// 不固的定键值对
type Type别名 = {
  [key: string]: 类型;
};
interface 接口名 {
  [key: string]: 类型;
}

                  // 固定的键值对
type TypeName1 = {
  name: string;
  age: number;
};
interface InterfaceName1 {
  name: string;
  age: number;
}

// 不固定的键值对
type TypeName2 = {
  [key: string]: string;
};
interface InterfaceName2 {
  [key: string]: string;
}

const b1: TypeName1 = { name: "x", age: 28 };
const b2: TypeName2 = { name: "x", age: "28" };
const c1: InterfaceName1 = { name: "x", age: 28 };
const c2: InterfaceName2 = { name: "x", age: "28" };

                  # 函数

                  三种写法：

                  1. 字面量（不建议）
                  2. type 类型别名 详见下文
                  3. interface 接口 详见跳转
                  • 写法一：字面量（不推荐）

                  结构不清晰，正经开发里别用

                  const a: () => void = () => console.log("hello");
const b: (text: string) => string = (text) => params;
                  • 写法二：type 类型别名
                  type FuncTypeA = {
  (): void;
};
const a: FuncTypeA = () => console.log("hello");

type FuncTypeB = {
  (text: string): string;
};
const b: FuncTypeB = (text) => text;
                  • 写法三：interface 接口
                  interface FuncA {
  (): void;
}
const a: FuncA = () => console.log("hello");

interface FuncB {
  (text: string): string;
}
const b: FuncB = (text) => text;

                  函数返回值的类型：

                  • void 类型
                    表示没有指定 return 返回值的函数
                    返回值为函数默认返回值 undefined
                  const func = (): void => console.log("hello");

console.log(func()); // undefined
                  • never 类型
                    void 类型的函数的返回值为 undefined，也算是个值
                    而 never 类型的函数是不会返回任何结果
                    比如：抛出异常的报错函数
                  const func = (): never => {
  throw new Error("出错了");
};

console.log(func());
// 无任何返回值，控制台（终端）报错，错误信息为 "出错了"
                  • any 类型（不要用）
                    详见上文
                  const a = (): any => {};
const b = (): unknown => {}; // 报错

                  剩余参数类型

                  const 函数名 = (参数: 类型, ...剩余参数: 元素的类型[]): 返回值类型 => {
  return 返回值;
};

                  # 联合・交叉类型

                  符号 含义 写法
                  联合类型 ｜ 或（满足一个即可） 类型 A ｜类型 B ｜类型 C
                  交叉类型 & 与（必须全部满足） 类型 A & 类型 B & 类型 C
                  • 类型定义为 联合类型 时，所赋值的类型只要属于其中即可
                  let varA: string | number;
varA = "1";
varA = 1;
varA = true; // 报错，不能将类型 boolean 分配给类型 string|number

let arrayB: Array<string | number>;
arrayB = [1, 2, 3];
arrayB = ["1", 2, "3"];
arrayB = ["1", false, "3"]; // 报错，不能将类型 boolean 分配给类型 string|number
                  • 交叉类型 多用于复杂类型
                    详见下文 type 的扩展
                  type name = {
  name: string;
};
type age = {
  age: number;
};
type person = name & age;

const Lili: person = { name: "Lili", age: 28 };
const tom: person = { name: "Tom" }; // 报错，缺少属性 age

                  # 类型定义

                  定义类型的写法：

                  1. 直接定义在数据后（不常用）
                  2. type 类型别名 详见下文
                  3. interface 接口 详见跳转

                  指定类型时通过 : 直接将类型放在数据后面

                  const 变量: 类型 = 值;

function 函数名(参数: 类型): 返回值类型 {
  return 返回值;
}
const 函数名 = (参数: 类型): 返回值类型 => 返回值;

                  重复使用或数据类型复杂时，建议使用 类型别名 type 或 接口 interface 单独定义类型。如下：

                  // 使用类型别名定义 Skill 类型
type Skill = {
  front?: string[];
  back?: string[];
};

// 使用接口定义 Person 类型，其中使用 Skill 类型
interface Person {
  name: string;
  age: number;
  skills?: Skill;
}

// 使用类型
const people: Person[] = [
  {
    name: "Andy",
    age: 28,
    skills: {
      front: ["React", "Vue", "Angular"],
      back: ["Node.js", "PHP", "Python", "Ruby"],
    },
  },
  { name: "Tom", age: 18 },
];

                  # 类型推论

                  若变量赋值时没定义类型时，TS 会根据赋予的值自动推论数据类型

                  let a = 10;
a = "你好"; // 报错，不能将类型 string 分配给类型 number

                  若定义变量时未赋值，则 TS 默认推论为 any 类型

                  let a;
a = 10;
a = "10";
a = null;
a = false;

                  # ❌类型断言

                  写法 说明
                  数据值 as 类型
                  <类型>数据值 不能用于 React 的 JSX

                  手动指定一个值的类型来覆盖 TS 的自动类型检测

                  类型断言不是类型转换，在编译结果中会被删除，不会真的影响到变量的类型

                  let a: unknown;
const b: string = a; // 报错
const c: string = a as string; // 不报错
                  • 将一个联合类型断言为其中一个类型
                  interface Cat {
  run(): void;
}
interface Fish {
  swim(): void;
}

function doSomething(animal: Cat | Fish) {
  // if (typeof animal.swim === 'function') return true;
  if (typeof (animal as Fish).swim === "function") return true;
  return false;
}
                  • 将一个父类断言为更加具体的子类
                  class ApiError extends Error {
  code: number = 0;
}
class HttpError extends Error {
  statusCode: number = 200;
}

function isApiError(error: Error) {
  // if (error instanceof ApiError) return true;
  if (typeof (error as ApiError).code === "number") return true;
  return false;
}
                  • 将 any 断言为一个具体的类型

                  第三方库未能定义好类型、项目历史遗留问题、别人写的烂代码、TypeScript 无法精确定义类型等情况时，不可避免的需要处理默认类型推论的 any 类型的变量

                  function getCacheData(key: string): any {
  return (window as any).cache[key];
}

interface Cat {
  run(): void;
}
const tom = getCacheData("tom") as Cat;
tom.run();

                  # 类型别名 ( type )

                  类型别名 type 定义类型的写法优于 上文所述 在数据后直接定义的写法

                  可实现类型重复使用，且类型复杂时使代码更清晰

                  类型别名 type 作用近似于 接口 interface，但扩展性弱于接口

                  # 定义

                  可用于声明任意类型（基础类型、对象类型、函数类型）

                  通过 = 赋值，别名首字母大写

                  同名 type 不能重复定义

                  // 基础类型
type BasicType = 类型;

// 对象类型
type ObjectType = {
  属性名: 类型;
  属性名: 类型;
};

// 函数类型
type FunctionType = (参数: 类型, 参数: 类型) => 返回值类型;

                  # 扩展

                  type 定义的类型内容的扩展（合并）使用 类型交叉 的 &

                  type A = 内容;
type B = 内容;

type C = A & B;

                  type Name = {
  name: string;
};
type Age = {
  age: number;
};
type Person = Name & Age;

const jack: Person = { name: "Jack", age: 28 };
const tom: Person = { name: "Tom" }; // 报错，缺少属性 age

                  type 也可以扩展合并 接口 interface

                  type A = 内容;

interface X {
  属性: 类型;
}

type B = A & X;

                  # 常用关键字

                  # ❌typeof

                  用于获取一个值的类型

                  type TypeName = typeof 目标值;

                  const A = 值;
const B: typeof A;

                  多用于复杂类型 的对象和数组

                  没必要 基本类型（直接字面量写就行了没必要再用 typeof）

                  const var1 = "hello";
const var2 = 1;
const var3 = true;
const var4 = { name: "tom", age: 20 };
const var5 = [1, 2, 3];

type T1 = typeof var1; // 无意义，相当于 type T1 = 'hello'
type T2 = typeof var2; // 无意义，相当于 type T2 = 1
type T3 = typeof var3; // 无意义，相当于 type T3 = true
type T4 = typeof var4; // 相当于 type T4 = { name: string, age: number }
type T5 = typeof var5; // 相当于 type T5 = number[]

                  const obj1 = {
  name: "tom",
  age: 20,
};

const obj2: typeof obj1 = {
  name: "jerry",
  age: 30,
};

                  # ❌keyof

                  用于获取一个类型中的所有 key 值组成的联合类型

                  type Person = {
  name: string;
  age: number;
};
const variable: keyof Person = "age"; // 相当于 const variable: 'name'|'age' = 'age'

                  // keyof 与 typeof 连用
const person = {
  name: "tom",
  age: 20,
};
const variable: keyof typeof person = "name"; // 相当于 const variable: 'name'|'age' = 'name'

                  # ❌in

                  用于遍历一个联合类型

                  type Keys = "a" | "b" | "c";

// { a: any, b: any, c: any }
type Obj = {
  [p in Keys]: any;
};

                  # ❌extends

                  用于类型继承

                  通常用于泛型约束和类型条件表达式

                  // 将泛型变量 T 约束为只能为 number 或 string 之一
type Func = <T extends number | string>(x: T) => T;

                  T extends U ? X : Y

                  # 类型条件表达式

                  interface Animal {
  live(): void;
}

interface Dog extends Animal {
  woof(): void;
}

// 结果为 type Example1 = number
type Example1 = Dog extends Animal ? number : string;

// 结果为 type Example2 = string
type Example2 = RegExp extends Animal ? number : string;
