            # JavaScript 的正则

            # 简介

            RegExp（Regular Expression）

            用于处理字符串

            是复杂数据类型

            console.log(typeof /abc/)
//#=> object

            # 作用

            • 用于验证字符串是否符合规则

            • 从字符串中获取一部分符合规则的内容

            使用特定的符号组合成一个表达式，来验证字符串或或取内容

            # 构成

            /正则元字符和字符/标识符

            写在双斜线中的是 正则表达式，写在双斜线外的是注释

            # 创建正则表达式

            # /abcd/

            字面量形式创建正则表达式

            const reg = /abc/
console.log(reg);

//#=> /abc/

            不能在里面直接写字符串变量

            let str = ‘hello’;
let reg = /str/
console.log(reg); 
//=>
/str/

            # new RegExp('abc')

            内置构造函数创建

            const reg = new RegExp('abc')
console.log(reg);
//#=> 
/abc/

            可以在里面直接写字符串变量

            let str = 'hello';
let reg = new RegExp(str);
console.log(reg);   
//#->
/hello/

            可以进行字符串拼接

            let str = 'hello';
let reg = new RegExp(str+' world');
console.log(reg); 
//#=>
/hello world/

            写入基础元字符时，为了和转义符\区分，要写 \\元字符

            let reg1 = /\w/
console.log(reg1);  
//
/\w/

let reg2 = new RegExp('\\w')
console.log(reg2);
//
/\w/

            # 正则的匹配与捕获

            # 匹配 test( )

            验证字符串是否符合正则规则

            	/abc/.test('abcd')

//  正则表达式.test(要检测的字符串)
//  返回值是布尔值 true / false

            什么位置、有几个都不重要，只要包含有就行，只要找到了第一个就行

            //检测是否包含一个abc片段

console.log(/abc/.test('abc'));   // true

console.log(/abc/.test('abcd'));  // true

console.log(/abc/.test('aabc'));  // true

console.log(/abc/.test('abcabc')); // true

console.log(/abc/.test('aabbcc')); // false

            # 捕获 exec( )

            从字符串中获取 符合正则规则的第一个片段

            	/abc/.exec('_abc_')

//  正则表达式.exec(检测的字符串)

            # 没有符合片段

            字符串中没有符合片段时，返回值时 null

            const res = /abc/.exec('zxc')
console.log(res);

//#=> 
null

            # 基础捕获

            字符串中有符合片段时，返回值是个数组

            const res = /abc/.exec('abcdddddeee')
console.log(res);

//#=>
[ 'abc', index: 0, input: 'abcdddddeee', groups: undefined ]

            [0] 是捕获的字符串，

            [1] 是个对象，index是指从字符串的第几个字符开始捕获的

            [2] 是个对象，input是原始字符串

            [ 'abc', 
 index: 0, 
 input: 'abcdddddeee',
 groups: undefined 
]

            即使字符串有多个符合的片段，只捕获字符串中符合的第一个片段

            const res1 = /abc/.exec('abcddddeee')
console.log(res1);

const res2 = /abc/.exec('abceeeeabc')
console.log(res2);

const res3 = /abc/.exec('eeeabceeabc')
console.log(res3);

//#=>
[ 'abc', index: 0, input: 'abcddddeee', groups: undefined ]
[ 'abc', index: 0, input: 'abceeeeabc', groups: undefined ]
[ 'abc', index: 3, input: 'eeeabceeabc', groups: undefined ]

            # 单独捕获与不捕获

            详见（）和（?:）

            ()里的正则即作为整体，在捕获时又被单独捕获

            (?:)里的正则只作为整体，不被捕获里面的内容

            # 有全局标识符g

            详见标识符

            当前捕获位置从上一次捕获位置后开始

            中间可以有其他内容

            直到找不到为止，返回null，然后再次开始捕获

            # 有粘性全局标识符y

            详见标识符

            必须字符串开始处就有符合的内容

            当前捕获位置紧跟上一次捕获位置

            中间不可以有其他内容，不然返回null然后结束捕获，然后再次开始捕获

            # \

            转义符，和字符串的转义符相同用法

            把没有意义的内容转为有意义的内容；

            把有意义的内容转为没有意义的内容

            比如：

            /s/  // 一个s字符

            /\s/ //一个空格

            /./  //一个非空格字符

            /\./  // 一个点字符

            /\\/  //一个斜线文本

            # -

            表示至，到

            从一个字符到另一个字符，比如0-9，a-z, A-Z

            		/0-255/

//0~2或5或5

            # 特殊字符和基础元字符

            		/[0-9]/
// 只要有0~9中的任意一个就可以
// 相当于 \d

            		/^[0-9]/
// 只要不是0~9中的任意一个就可以
// 相当于 \D

            		/^[0-9]$/
// 只能由一个字符
// 只能是0~9中的任何一个

            		/[0-9a-zA-Z_/
// 只要有0-9，a-z,  A-Z ，和下划线任意一个就行

            # 正向否定预查（?!）

            在捕获一个内容时，该内容的后面必须不能跟有选择的某项

            		 /捕获内容(?!内容后跟的选择)/

            与（?=）相反，正向肯定预查是要求捕获内容后跟有某项

            （?!）正向否定预查是要求捕获内容后不能跟有

            如下：

//（捕获的内容是ES，ES后必须跟有2015|2016）

            		/ES(?!2017)/
//只要ES后跟的不是2017，就捕获这个字符串中的ES

        let arr = [];

        //先捕获一次
        let tmp = reg.exec(str);

        //因为捕获的结果 ，要么捕获成功返回一个数组，要么没捕获到返回null

        while (tmp) {
            arr.push(tmp[0])

            tmp = reg.exec(str)
        };

        console.log(arr);
