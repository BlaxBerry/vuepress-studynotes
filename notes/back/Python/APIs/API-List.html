# List 常用方法

# 增加

# append()

在列表尾部添加一个元素

无返回值，会修改原来的列表

列表.append(元素)

如下：

example = [1, 2, 3]

example.append(4)

print(example)
# [1, 2, 3, 4]

# extend()

在列表末尾追加一组元素

无返回值，会修改原来的列表

列表.append(一组数据序列)

追加的数据序列可为列表、元组、集合、字典

若为字典则仅将 key 键 依次添加为元素

如下：

example = [1, 2, 3]

# 列表的场合
example.extend( [4, 5, 6])
print(example)  # [1, 2, 3, 4, 5, 6]

# 元组的场合
example.extend((4, 5, 6))
print(example)  # [1, 2, 3, 4, 5, 6]

# 集合的场合
example.extend({4, 5, 6})
print(example)  # [1, 2, 3, 4, 5, 6]

# 字典的场合
example.extend({'key': 'value'})
print(example)  # [1, 2, 3, 'key']

# insert()

将指定对象插入列表的指定序号位置

修改原列表

列表.insert(序号, 新元素)

如下：

example = ['andy', 'jack', 'tom']

example.insert(1, 'lili')

print(example)
# ['andy', 'lili', 'jack', 'tom']

# 删除

# pop()

从列表中删除指定序号的元素

参数默认序号为**-1**，删除列表最后一个元素

会改变列表

返回值是删除的元素

# 删除最后一个元素
列表.pop()

# 删除指定序号元素
列表.pop(序号)

删除的元素 = 列表.pop()

如下：

example_list = [1, 2, 3, 4, 5]

a = example_list.pop()
b = example_list.pop(2)

print(example_list)
# [1, 2, 4]

print(a)  # 5
print(b)  # 3

# remove()

从列表中删除指定元素

无返回值，会修改原列表

列表.remove(元素)

如下：

example_list = [1, 2, 3, 4, 5]

a = example_list.remove(2)
b = example_list.remove(3)

print(example_list)
# [1, 4, 5]

print(a)  # None
print(b)  # None

# clear()

清空列表内所有元素

列表.clear()

如下：

list = ['andy', 'tom', 'jack']

list.clear()

print(list) # []

# 查找

# index()

查找元素在列表中第一次出现位置的序号

序号 = 列表.index(元素)

如下：

example = ['andy', 'jack', 'tom', 'jack']

index = example.index('tom')
print(index)  # 2

# 统计

# len(List)

获取列表的长度，即元素个数

元素个数 = len(列表)

如下：

list1 = [0, 1, 2, 3, 4]

length = len(list1)

print(length)  # 5

# count

获取某元素在列表中的出现次数

列表.count(元素)

如下：

example_list = [3, 2, 2, 4, 5, 2]

count = example_list.count(2)

print(count)  # 3

# 排序

# sort()

列表排序

# 升序排列
列表.sort()

# 降序排列
列表.sort(reverse=True)

如下：

example_list = [3, 4, 1,  3, 5, 2]

example_list.sort()
print(example_list)
# [1, 2, 3, 4, 5]

example_list.sort(reverse=True)
print(example_list)
# [5, 4, 3, 2, 1]

# reverse()

反转列表

列表.reverse()

如下：

example_list = [3, 4, 1,  3, 5, 2]

example_list.reverse()
print(example_list)
# [2, 5, 3, 1, 4, 3]

# 复制

# copy()

复制列表，类似于 列表[:]

是浅拷贝，复制出来的列表在内存中位置不同于原始列表

列表 = 列表.copy()

如下：

a = [1, 2, 3]

b = a.copy()
print(id(b) == id(a))  # False

c = a[:]
print(id(c)==id(a))  # False

d = a
print(id(d) == id(a))  # True

# 转换

# list()

列表 = list(数据列表)

如下：将元组转换为列表

tup = (1, 2, 3, 4)
print(type(tup))  # <class 'tuple'>

lis = list(tup)
print(type(lis))  # <class 'list'>
print(lis)  # [1, 2, 3, 4]
