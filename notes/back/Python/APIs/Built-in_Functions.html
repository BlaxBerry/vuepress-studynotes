# 输入输出

# print()

打印参数到控制台

多个参数时用逗号隔开，逗号不会被打印出来

print(参数)
print(参数, 参数)
print(参数, end='')
print('hello, ', 'I am ', 28, 'years old')
# 'hello,  I am  28 years old'

默认最后一个参数 end='/n'，即默认输出完后另起一行

for item in range(1, 4):
    print(item)
# 1
# 2
# 3


for item in range(1, 4):
    print(item, end=' ')
# 1 2 3

# 长度

# len()

获取有长度的对象的长度

print(len('hello'))  # 5
print(len([1, 2, 3]))  # 3
print(len((1, 2, 3)))  # 3
print(len({'name': 'Andy', 'age': 28}))  # 2
print(len({1, 2, 3}))  # 3

# 计算

# max()

获取一组数据中的最大值

l = [1, 2, 3]
t = (1, 2, 3)
st = {1, 2, 3}

print(max(l))  # 3
print(max(t))  # 3
print(max(st))  # 3

# min()

获取一组数据中的最小值

l = [1, 2, 3]
t = (1, 2, 3)
st = {1, 2, 3}

print(min(l))  # 1
print(min(t))  # 1
print(min(st))  # 1

# sum()

一组数据求和

l = [1, 2, 3]
t = (1, 2, 3)
st = {1, 2, 3}

print(sum(l))  # 6
print(sum(t))  # 6
print(sum(st))  # 6

# abs()

绝对值

print(abs(0))  # 0
print(abs(1))  # 1
print(abs(-1))  # 1

# 格式化

# sorted()

序列排序

其他类型数据可不再借助列表的 sort() 方法

如下：元组排序

t = (2, 3, 1, 4, 5)

print(sorted(t))  # [1, 2, 3, 4, 5]
print(sorted(t, reverse=True))  # [5, 4, 3, 2, 1]

如下：集合排序

st = {2, 3, 1, 4, 5}

print(sorted(st))  # [1, 2, 3, 4, 5]
print(sorted(st, reverse=True))  # [5, 4, 3, 2, 1]

# 转换

# chr()

获取 ASCII 码对应的字符

print(chr(97))  # a
print(chr(65))  # A

# ord()

获取字符对应的 ASCII 码

print(ord('A'))  # 65
print(ord('a'))  # 97
