# 输入输出

# print()

打印参数到控制台

多个参数时用逗号隔开，逗号不会被打印出来

print(参数)
print(参数, 参数)
print(参数, end='')
print('hello, ', 'I am ', 28, 'years old')
# 'hello,  I am  28 years old'

默认最后一个参数 end='/n'，即默认输出完后另起一行

for item in range(1, 4):
    print(item)
# 1
# 2
# 3


for item in range(1, 4):
    print(item, end=' ')
# 1 2 3

# 长度

# len()

# 计算

# max()

获取一组数据中的最大值

最大值 = max(一组数)

# 按匿名函数返回值表达式值获取最大值
最大值 = max(一组数据, key=lambda 元素: 元素表达式)

# 列表
print(max([1, 2, 3]))  # 3
# 元组
print(max((1, 2, 3)))  # 3
# 集合
print(max({1, 2, 3}))  # 3

如下：获取字典元素列表中某一属性的最值

ls = [
    {'name': 'Andy', 'age': 28},
    {'name': 'Jack', 'age': 17},
    {'name': 'Tom', 'age': 30}
]

# age 最大的元素
target = max(ls, key=lambda x: x['age'])
print(target)  # {'name': 'Tom', 'age': 30}

# 排序

# sorted()

排序一组可迭代序列

# 默认升序
列表 = sorted(一组数据, reverse=False)
# 降序
列表 = sorted(一组数据, reverse=True)

# 按匿名函数返回值表达式值进行排列
列表 = sorted(一组数据, key=lambda 元素: 元素表达式)

# 元组
t = (2, 3, 1, 4, 5)
print(sorted(t))  # [1, 2, 3, 4, 5]
print(sorted(t, reverse=True))  # [5, 4, 3, 2, 1]

# 集合
st = {2, 3, 1, 4, 5}
print(sorted(st))  # [1, 2, 3, 4, 5]
print(sorted(st, reverse=True))  # [5, 4, 3, 2, 1]

如下：按照字典元素列表中某一属性升降序排列

ls = [
    {'name': 'Andy', 'age': 28},
    {'name': 'Jack', 'age': 17},
    {'name': 'Tom', 'age': 30}
]

# 按 age 升序排列
l_up = sorted(ls, key=lambda x: x['age'])

for item in l_up:
    print(item)

# {'name': 'Jack', 'age': 17}
# {'name': 'Andy', 'age': 28}
# {'name': 'Tom', 'age': 30}


# 按 age 降序排列
l_down = sorted(ls, key=lambda x: x['age'], reverse=True)

for item in l_down:
    print(item)

# {'name': 'Tom', 'age': 30}
# {'name': 'Andy', 'age': 28}
# {'name': 'Jack', 'age': 17}
