# Str 常用方法

# 去空白

# strip()

去字符串左右两端空格

是 lstrip() + rstrip() 的复合写法

str1 = " aaaa " 

print(str1)  # " aaaa "
print(str1.strip())  # "aaaa"

# 文本对齐

# center()

将字符串填充至指定长度并使内容居中展示

返回的是一个新字符串

若指定长度小于原字符串长度，则返回原字符串

第二个参数是充填字符，默认使用英文半角空格字符，可省略

新字符串 = 字符串.center(长度)

新字符串 = 字符串.center(长度, "充填字符")

如下：

str1 = "你好"

print(str1.center(6))  # "  你好  "
print(str1.center(6, "+"))  # "++你好++"

example = ["咏鹅", "骆宾王", "鹅鹅鹅", "曲项向天歌", "白毛浮绿水", "红掌拨清波"]

for item in example:
    print(f"|{item.center(5,'〇')}|")

"""
|〇〇咏鹅〇|
|〇骆宾王〇|
|〇鹅鹅鹅〇|
|曲项向天歌|
|白毛浮绿水|
|红掌拨清波|
"""

# ljust()

将字符串填充至指定长度并使内容左对齐展示

其实就是默认的展示方式

第二个参数是充填字符，默认使用一个英文半角空格字符，可省略

返回的是一个新字符串

新字符串 = 字符串.ljust(长度)

新字符串 = 字符串.ljust(长度, "充填字符")

如下：

example = ["咏鹅", "骆宾王", "鹅鹅鹅", "曲项向天歌", "白毛浮绿水", "红掌拨清波"]

    print(f"|{item.ljust(5,'〇')}|")

"""
|咏鹅〇〇〇|
|骆宾王〇〇|
|鹅鹅鹅〇〇|
|曲项向天歌|
|白毛浮绿水|
|红掌拨清波|
"""

# 大小写

# title()

将每个单词的首字母都转化为大写

其余字母都转为小写

返回值是处理后的字符串

新字符串 = 字符串.title()

s = "hEllo, mY namE iS Andy"

t = s.title()
print(t)
# "Hello, My Name Is Andy"

非字母后的第一个字母都转为大写

如下：

s = "b2b2b2 and 3g3g3g"

t = s.title()
print(t)
# "B2B2B2 And 3G3G3G"

# capitalize()

将字符串的第一个字母变成大写，其他字母变小写

新字符串 = 字符串.title()

如下：

a = 'hello, my name is Andy'

b = a.capitalize()
print(b)
# "Hello, my name is andy"

若首字符是非字母，则字符串字母全转换成小写

如下：

a = '123hello, Andy'

b = a.capitalize()
print(b)
# "123hello, andy"

# 判断内容

# isspace()

判断字符串是否只包含空格或空白字符

返回值为布尔型 True 或 False

布尔 = 字符串.isspace()

如下：

str1 = " "
print(str1.isspace())  # True

str2 = "hello hello"
print(str2.isspace())  # False

str3 = ""
print(str3.isspace())  # False

str4 = "\n \t \r"
print(str4.isspace())  # True

# isalnum()

判断字符串是否由包含一个以上的字母或数字组成

返回值为布尔型 True 或 False

布尔 = 字符串.isalmun()

如下：

str1 = "1"
print(str1.isalnum())  # True

str2 = "hello123"
print(str2.isalnum())  # True

str3 = "hello"
print(str3.isalnum())  # True

str4 = "   "
print(str4.isalnum())  # False

str5 = ""
print(str5.isalnum())  # False

# isalpha()

print(str2.isalpha())  # False

print(str3.isalpha()) # False

print(str4.isalpha())  # False

str5 = "你好"
print(str5.isalpha())  # True

str6 = "こんにちは"
print(str6.isalpha())  # True

# isdecimal()

判断字符串是否只由整数数字构成

仅限十进制阿拉伯数字

返回值为布尔型 True 或 False

如下：

str1 = "123"

print(str3.isdecimal())  # False

# 查找

# find()

返回指定字符或子字符串在字符串中的索引序号

若不存在否则返回 -1

序号 = 字符串.find(查找字符)

序号 = 字符串.find(查找字符, beg=开始序号, end=结束序号)

参数 start、end 用来指定查找范围，可以省略

  • beg：开始索引，默认是 0
  • end：结束索引，默认是 len(字符串)

如下：

s = "你好，我是Andy"

print(s.find("Andy"))  # 5
print(s.find("Bill"))  # -1

# index()

用法与find()方法一样

返回指定字符或子字符串在字符串中的索引序号

但 找不到指定字符时是报错返回一个异常

ValueError: substring not found

序号 = 字符串.index(查找字符)

序号 = 字符串.index(查找字符, beg=开始序号, end=结束序号)

如下：

s = "你好，我是Andy"

print(s.index("Andy"))  # 5
print(s.index("Bill"))  # 报错

# count()

获取字符或子字符串在字符串中出现的次数

若不存在则返回 0

次数 = 字符串.count(字符)

次数 = 字符串.count(字符, start=开始序号, end=结束序号)

参数 start、end 用来指定查找范围，可以省略

  • start：开始序号，默认为 0
  • end：结束序号，默认为 len(字符串)

如下：

str1 = 'banana pie banana'

print(str1.count('a'))  # 6
print(a.count('你好'))  # 0

# 替换

# replace()

将字符串中的指定字符全部替换

返回值是处理后的新字符串

新字符串 = 字符串.replace('字符', '替换字符')

新字符串 = 字符串.replace('字符', '替换字符', 仅替换前几个)

如下：

string = "傻逼打野，你是个大傻逼"

str_handled = string.replace("傻逼", "**")
print(str_handled)
# "**打野，你是个大**"

str_handled = string.replace("傻逼", "**", 1)
print(str_handled)
# "**打野，你是个大傻逼"

string = "智障! 傻逼打野，你是个大傻逼，你就是个智障"

badwords = ["傻逼", "智障"]

for word in badwords:
    string = string.replace(word, "**")

print(string)
# "**! **打野，你是个大**，你就是个**"

# 转换

# split()

按间隔符将字符串分割为字符串列表

若间隔符不存在，则默认将空白字符(\r\n\t)分割

列表 = 字符串.split()

列表 = 字符串.split("分割字符")

列表 = 字符串.split("分割字符", 分割到第几个)

如下：

str1 = "1 2 3 4"

print(str1.split())
# ['1', '2', '3', '4']

print(str1.split(" "))
# ['1', '2', '3', '4']

print(str1.split("+++"))
# ['1 2 3 4']

print(str1.split(" ", 2))
# ['1', '2', '3 4']

# join()

将序列(列表、元组、字典)的元素连接上间隔符合并为一个字符串

字符串 = "分隔符".join(列表)

如下：

str1 = "咏鹅\n骆宾王\n鹅鹅鹅\n曲项向天歌\n白毛浮绿水\n红掌拨清波"

print("".join(str1.split()))
# "咏鹅骆宾王鹅鹅鹅曲项向天歌白毛浮绿水红掌拨清波"
