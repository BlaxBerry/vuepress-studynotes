          # 简介

          集合(Set)是一组无序不重复的数据组成的序列

          底层是通过字典的 key 键唯一特性实现

          列表 与 集合：

          List：有序，有下标，元素可以重复

          Set：无序，无下标，元素不重复


          # 创建

          集合通过 set() 方法创建

          st =set()

print(st)  # set()
print(type(st))  # <class 'set'>

st1 =set({1, 2, 3})
print(st1)  # {1, 2, 3}

          也可通过大括号 {} 创建

          但空集合必须通过 set() 方法创建，否则就成字典了

          st = {1, 2, 3, 2, 2}

print(st)  # {1, 2, 3}
print(type(st))  # <class 'set'>

          如下：列表去重复

          def get_unique(params):

    return list(set(params))


print(get_unique([1, 2, 3, 2, 2]))
print(get_unique([9, 9, 6]))

          # 集合推导式

          { 表达式 for 表达式 in 列表 }

          l = [1, 2, 3, 2, 2]

s = {item for item in l}
print(s)  # {1, 2, 3}

print({item * 10 for item in l})
# {10, 20, 30}

print({item for item in l if item % 2 == 0})
# {2}

          # 添加元素

          # add()

          重复元素不会被添加

          集合.add(元素)

          st = set()
print(st)  # set()

st.add(1)
print(st)  # {1}

st.add(1)
print(st)  # {1}

st.add(2)
print(st)  # {1, 2}

st.add(3)
print(st)  # {1, 2, 3}

          # update()

          添加多个元素（合并集合）

          集合.add({元素, 元素, 元素})
集合A.update(集合B)

          st = set({1, 2, 3})
print(st)  # {1, 2, 3}

st.update({4, 5, 6})
print(st)  # {1, 2, 3, 4, 5, 6}

          # 移除元素

          # remove()

          从集合中移除指定元素

          若元素不存在则报错，抛出一个 keyError 异常

          集合.remove(指定元素)

          st = {1, 2, 3, 4}

st.remove(2)
print(st)  # {1, 3, 4}

st.remove(999) # 报错 KeyError: 999

          # discard()

          从集合中移除指定元素

          元素不存在就不删了，不报错

          集合.discard(指定元素)

          st = {1, 2, 3, 4}

st.discard(999)  # 找不到元素，do nothing

          # clear()

          清空集合内容

          集合.clear()

          st = {1, 2, 3, 4}

st.clear()
print(st)  # set()

st.add(1)
print(st)  # {1}

          # 判断元素

          # issubset()

          判断一个集合是否为另一集合的子集（包含在内）

          # 判断 A 是否为 B 的子集
A.issubset(B)

# 判断 B 是否为 A 的子集
B.issubset(A)

          st1 = {3, 4}
st2 = {1, 2, 3, 4}
st3 = {1, 2, 3, 4}

print(st1.issubset(st2))  # True
print(st2.issubset(st1))  # False
print(st2.issubset(st3))  # True

          # 并集

          加到一起的

          # A | B

          集合A | 集合B

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1 | st2)  # {1, 2, 3, 4, 5, 6}

          # union()

          集合A.union(集合B)

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1.union(st2))  # {1, 2, 3, 4, 5, 6}

          # 交集

          彼此都有重合的部分

          # A & B

          集合A & 集合B

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1 & st2)  # {3, 4}

          # intersection()

          集合A.intersection(集合B)

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1.intersection(st2))  # {3, 4}

          # 差集

          各自独有的部分

          # A - B 与 B - A

          # A 有 B 没有的
集合A - 集合B

# B 有 A 没有的
集合B - 集合A

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1 - st2)  # {1, 2}
print(st2 - st1)  # {5, 6}

          # difference()

          # A 有 B 没有的
集合A.difference(集合B)

# B 有 A 没有的
集合B.difference(集合A)

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1.difference(st2))  # {1, 2}
print(st2.difference(st1))  # {5, 6}

          # 对称差集

          相互不重复的部分

          # A ^ B

          集合A ^ 集合B

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1 ^ st2)  # {1, 2, 5, 6}
print(st2 ^ st1)  # {1, 2, 5, 6}

          # symmetric_difference()

          集合A.symmetric_difference(集合B)

          st1 = {1, 2, 3, 4}
st2 = {3, 4, 5, 6}

print(st1.symmetric_difference(st2))  # {1, 2, 5, 6}
print(st2.symmetric_difference(st1))  # {1, 2, 5, 6}

          # 使用场景

          如下：生成 5 组不重复的 4 位的随机验证码

          import random

final_list = set()
chars = 'ABCDEFGHIJKLMNOPQRSTUVWSYZ'

while True:
    code = ''
    for i in range(4):
        code += random.choice(chars)

    final_list.add(code)

    if len(final_list) == 5:
        break


print(final_list)
