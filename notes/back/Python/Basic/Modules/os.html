# 常用方法

os 模块用于文件和目录操作管理

import os
import os.path

# rename()

重命名文件

os.rename('旧文件路径', '新文件名')

import os

os.rename('./02.txt', '02.json')

重命名目录

os.rename('旧目录路径', '新目录名')

import os

os.rename('./dir', 'haha')

# remove()

删除文件

os.remove('文件路径')

import os

os.remove('./02.json')

# listdir()

以列表形式获取目录列表

os.listdir('目录地址')

import os

print(os.listdir('.'))
print(os.listdir(os.getcwd()))

# mkdir()

创建目录

os.mkdir('目录路径')

import os

os.mkdir('dir')

# rmdir()

删除一个非空目录

os.rmdir('目录路径')

import os

os.rmdir('dir')

若目录不为空则抛出 OSError 异常

# OSError: Directory not empty

# getcwd()

获取当前目录绝对路径

os.getcwd()

import os

print(os.getcwd())

# os.chdir()

改变当前工作目录

os.chdir('新的工作目录')

import os

print (os.getcwd())

os.chdir( path )
print (os.getcwd())

# os.path 模块

import os.path

# isdir()

判断是否是目录

import os.path

print(os.path.isdir('01.py'))

# isfile()

判断是否是文件

import os.path

print(os.path.isfile('01.py'))

# splitext()

以元组形式分隔文件名与后缀

os.path.splitext('文件名.后缀名')

import os.path

portion = os.path.splitext('file_name.txt')
print(portion)  # ('file_name', '.txt')
print(portion[0])  # file_name
print(portion[1])  # .txt

# abspath()

返回绝对路径

import os.path

print(__file__)
# /Users/xxx/StudyPractice/Python/02.py

print(os.path.abspath('./02.py'))
# /Users/xxx/StudyPractice/Python/02.py

# basename()

获取路径中的文件名

import os.path

print(os.path.basename("/StudyPractice/Python/02.py"))
# 02.py

print(os.path.basename(__file__))
# 02.py

# dirname()

获取路径中文件所在目录

import os.path

print(os.path.dirname("/StudyPractice/Python/02.py")
# /StudyPractice/Python

# abspath()

返回绝对路径

import os.path

print(os.path.abspath('./01.py'))
# /Users/xxx/StudyPractice/Python/01.py

# split()

以元组形式分隔路径中所在目录与文件名

import os.path

print(os.path.split('/StudyPractice/Python/02.py'))
# ('/StudyPractice/Python', '02.py')

# join()

合并路径

import os.path

print(os.path.join('a', 'b', 'file.txt'))
# a/b/file.txt

# 修改目录下文件后缀

import os
import os.path

def change_extension(
    target_dir: str,
    ext_from: str,
    ext_to: str
):
    all_files = os.listdir(target_dir)
    os.chdir(target_dir)

    for filename in all_files:
        portion = os.path.splitext(filename)
        if portion[1] == ext_from:
            newname = portion[0] + ext_to
            os.rename(filename, newname)


change_extension('./dir', '.txt', '.json')

# before:
|- dir
	|- 01.txt
	|- 02.txt
|- demo.py

# after:
|- dir
	|- 01.json
	|- 02.json
|- demo.py
