    # return

    可通过 return 关键字设置函数的返回值

    若没有 return 返回值则默返回 None

    如下：有返回值

    def example_func():

    return '你好'

res = example_func()
print(res)  # '你好'

    如下：无返回值

    def example_func():

    print('你好')


res = example_func()
print(res)  # None

    return 后的代码不会执行

    可直接用来结束函数的执行

    def example_func():

  	print('aaaa')

    return

    print('bbbb')

    # 多个参数

    形参按 顺序接收实参传递的数据

    函数调用时传递的实参的个数必须与形参一致否则程序会报错

    def fn(name, age):

    print(f'name: {name}')
    print(f'age: {age}')


# 个数不一致
fn('andy')  # 报错

# 按顺序传参
fn('andy', 28)
# name: andy
# age: 28

# 不按顺序传参
fn(28, 'andy')
# name: 28
# age: andy

  pass

    如下：

    # 假设男性居多的场合可直接设置默认性别
def show_person(name, is_male=True):

    if is_male:
        gender = 'male'
    else:
        gender = 'female'

    return ({
        "name": name,
        'gender': gender
    })


print(show_person('Andy'))
# {'name': 'Andy', 'gender': 'male'}

print(show_person('Lili', is_male=False))
# {'name': 'Lili', 'gender': 'female'}

print(show_person('Any', False))
# {'name': 'Any', 'gender': 'female'}

    参数默认值多用于该值为常用的场合

    比如 sort() 方法的参数 reverse 默认值为 False

    不传入参数方法函数就使用默认值，但若传入参数比如指定 reverse=True 则方法函数采用 传入值

    arr = [2, 4, 1, 3]

arr.sort()
print(arr)  # [1, 2, 3, 4]

arr.sort(reverse=False)
print(arr)  # [1, 2, 3, 4]

arr.sort(reverse=True)
print(arr)  # [4, 3, 2, 1]

    # 元组与字典拆包

    在函数接收多个参数时

    若想将一个元组直接传递给 args，实参前要带上 *

    若想将一个字典直接传递给 kwargs，实参前要带上 **

    def func(*args, **kwargs):

    print(args)
    print(kwargs)


g_tuple = (1, 2, 3,)
g_dict = {'k1': 1, 'k2': 2}

func(*g_tuple, **g_dict)
# (1, 2, 3)
# {'k1': 1, 'k2': 2}

    若不使用拆包，则需要拆开依次传递，太麻烦

    def func(*args, **kwargs):

    print(args)
    print(kwargs)


func(1, 2, 3, k1=1, k2=2)
# (1, 2, 3)
# {'k1': 1, 'k2': 2}

    # 不可变类型

    函数内部直接通过赋值语句修改接收的参数引用时

    被视作在函数内部定义了与参数同名的局部变量并引用赋值的内存地址

    此时的赋值修改不会影响外部调用时传入的实参

    def func(params):

    print(params)  # 1
    params = 2
    print(params)  # 2


a = 1
func(a)

print(a)  # 1

    # 可变类型

    若函数调用时传入的实参时可变类型数据（列表、字典）

    内部通过对应方法修改参数引用时会影响外部传入的实参

    def func(params):

    print(params)  # [1,2,3]
    params.pop()
    print(params)  # [1,2]


a = [1, 2, 3]
func(a)

print(a)  # [1,2]

    # 递归

    递归是指一个函数在其内部自己调用自己

    因为是自己调用自己，递归只是针对不同的参数进行不同处理

    必须设置有一个结束条件，以防止调用出现死循环

    默认最大递归深度不超过 1000 次

    如下：利用递归循环打印参数直至条件满足

    def example_func(params):

    if params > 5:
        return

    print(params)

    example_func(params+1)


example_func(1)
# 1
# 2
# 3
# 4
# 5

    # 实例 - 累加

    如下：

    递归累加可理解为数学的 f(x) = x+f(x-1)

    def get_sum(end):
    """计算 1 + 2 + ... + num 的累加和"""
    if end == 1:
        return 1

    next = get_sum(end-1)

    return next + end

print(get_sum(1))  # 1
print(get_sum(2))  # 3
print(get_sum(3))  # 6
print(get_sum(100))  # 5050

    # 实例 - 斐波那契数列

    :: tip 斐波那契数列特点：

    1, 1, 2, 3, 5, 8, 13, 21, ... a, b, c, d, e, .....

    a + b = c b + c = d d + e = f ....

    ::

    def get_fibo(index):
    """递归获取指定位置的数字"""
    if index == 0:
        return 0
    elif index == 1 or index == 2:
        return 1
    else:
        return get_fibo(index-2) + get_fibo(index-1)


print(get_fibo(1))  # 1
print(get_fibo(2))  # 1
print(get_fibo(3))  # 2
print(get_fibo(4))  # 3
print(get_fibo(5))  # 5



def show_list(end_index):
    """获取列表形式"""
    final_list = []

    for i in range(end_index):
        final_list.append(get_fibo(i))

    return final_list


print(show_list(8))
# [0, 1, 1, 2, 3, 5, 8, 13]
