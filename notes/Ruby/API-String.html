# 字符串常用API

# .length

# .size

字符长度

p ''.length			# 0
p 'hello'.length # 5

obj={
    name:'andy',
    age:28
}

p obj.to_a
# [[:name, "andy"], [:age, 28]]

#

# .to_s

强制转为字符串

数据.to_s

n = 100
p n.to_s
# "100"

arr = [1,2,3]
p arr.to_s
# "[1, 2, 3]"

hash = {name: "andy"}
p hash.to_s
# "{:name=>\"andy\"}"

#

# .empty?

判断字符串是否是空

返回 true / false

字符串.empty?

空格也算内容不算是空

p ''.empty?		# true
p ' '.empty?	# false
p 'a'.empty?	# false

# .include?( )

判断字符串中过是否包含某字符

返回 true / false

字符串.include?('字符串')

p "helloruby".include?('ruby')	# true

# .chop

删除字符串的最后一个字符

返回一个新字符串，不修改原字符串

新字符串 = 字符串.chop

str = "ruby"

s = str.chop
p s
# "rub"
p str
# "ruby"

# .chop!

.chop：返回一个新字符串，不修改原字符串

.chop!：修改原字符串

字符串.chop

str = "ruby"

str.chop!
p str
# "rub"

# .chomp

移除字符串末尾的换行字符 \n

返回一个新字符串，不修改原字符串

若字符串末尾没有\n，则不进行任何操作

新字符串 = 字符串.chomp!

str = "ruby\n"

s = str.chomp
p s
# "ruby\n"
p str
# "ruby"

# .chomp!

.chomp：返回一个新字符串，不修改原字符串

.chomp!：修改原字符串

字符串.chomp!

str = "ruby\n"

str.chomp!
p str
# "ruby"

# .upcase

所有字母转为大写

返回一个新字符串，不修改原字符串

新字符串 = 字符串.upcase

str = "ruby"

s = str.upcase

p s		# "RUBY"
p str # "ruby"

# .upcase!

.upcase：返回一个新字符串，不修改原字符串

.upcase!：修改原字符串

字符串.upcase!

str = "ruby"

str.upcase!

p str	# "RUBY"
